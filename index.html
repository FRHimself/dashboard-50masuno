set.seed(123)
provincias <- c("Lima","Arequipa","Cusco","Piura","La Libertad","Puno","Ayacucho","Junín","Ica","Loreto")
apariciones <- data.frame(
  provincia = provincias,
  apariciones_2025 = sample(5:40, length(provincias), replace = TRUE)
)

url <- "https://raw.githubusercontent.com/johan/world.geo.json/master/countries/PER.geo.json"
peru <- st_read(url, quiet = TRUE)
peru$provincia <- sample(provincias, nrow(peru), replace = TRUE)

data_map <- left_join(peru, apariciones, by = "provincia")

gg <- ggplot(data_map) +
  geom_sf(aes(fill = apariciones_2025), color = "white") +
  scale_fill_viridis_c(option = "C") +
  labs(title = "Apariciones públicas de Rafael López Aliaga por provincia (2025)",
       fill = "Apariciones") +
  theme_minimal()

ggplotly(gg)

Análisis del Discurso

library(plotly)
mes <- month.abb[1:10]
df <- data.frame(
  mes=factor(mes, levels=mes),
  conservadurismo=runif(10,40,80),
  proteccionismo=runif(10,20,70),
  estado_derecho=runif(10,30,95),
  libertad_prensa=runif(10,10,60)
)
plot_ly(df, x=~mes) %>% 
  add_lines(y=~conservadurismo, name="Conservadurismo") %>%
  add_lines(y=~proteccionismo, name="Proteccionismo") %>%
  add_lines(y=~estado_derecho, name="Estado de Derecho") %>%
  add_lines(y=~libertad_prensa, name="Libertad de Prensa") %>%
  layout(title="Evolución del Discurso (2025)", hovermode="x unified")
